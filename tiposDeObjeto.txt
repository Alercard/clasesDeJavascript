// Declaracion de un objeto
var p1 = {x: 0, y:3 }
var p2 = {x: 4, y:0 }
function distancia(p1, p2) {
   const x=p1.x - p2.x;
   const y=p1.y - p2.y;
   return Math.sqrt(x*x + y*y);
}
console.log(distancia(p1,p2));

console.log(
   distancia(
      p1,
      /* objeto an√≥nimo */
      {x:0, y:7}
   )
);
